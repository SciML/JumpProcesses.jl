544 """
545 check_error(integrator)
546
547 Check state of `integrator` and return one of the
548 [Return Codes](https://docs.sciml.ai/DiffEqDocs/stable/basics/solution/#retcodes)
549 """
550 function check_error(integrator::DEIntegrator)
551 if integrator.sol.retcode âˆ‰ (ReturnCode.Success, ReturnCode.Default)
552 return integrator.sol.retcode
553 end
554 opts = integrator.opts
555 verbose = opts.verbose
556 # This implementation is intended to be used for ODEIntegrator and
557 # SDEIntegrator.
558 if isnan(integrator.dt)
559 if verbose
560 @warn("NaN dt detected. Likely a NaN value in the state, parameters, or derivative value caused this outcome.")
561 end
562 return ReturnCode.DtNaN
563 end
564 if integrator.iter > opts.maxiters
565 if verbose
566 @warn("Interrupted. Larger maxiters is needed. If you are using an integrator for non-stiff ODEs or an automatic switching algorithm (the default), you may want to consider using a method for stiff equations. See the solver pages for more details (e.g. https://docs.sciml.ai/DiffEqDocs/stable/solvers/ode_solve/#Stiff-Problems).")
567 end
568 return ReturnCode.MaxIters
569 end
570
571 # The last part:
572 # Bail out if we take a step with dt less than the minimum value (which may be time dependent)
573 # except if we are successfully taking such a small timestep is to hit a tstop exactly
574 # We also exit if the ODE is unstable according to a user chosen callback
575 # but only if we accepted the step to prevent from bailing out as unstable
576 # when we just took way too big a step)
577 step_accepted = !hasproperty(integrator, :accept_step) || integrator.accept_step
578 if !opts.force_dtmin && opts.adaptive
579 if abs(integrator.dt) <= abs(opts.dtmin) &&
580 (!step_accepted || (hasproperty(opts, :tstops) ?
581 integrator.t + integrator.dt < integrator.tdir * first(opts.tstops) :
582 true))
583 if verbose
584 if isdefined(integrator, :EEst)
585 EEst = ", and step error estimate = $(integrator.EEst)"
586 else
587 EEst = ""
588 end
589 @warn("dt($(integrator.dt)) <= dtmin($(opts.dtmin)) at t=$(integrator.t)$EEst. Aborting. There is either an error in your model specification or the true solution is unstable.")
590 end
591 return ReturnCode.DtLessThanMin
592 elseif !step_accepted && integrator.t isa AbstractFloat &&
593 abs(integrator.dt) <= abs(eps(integrator.t))
594 if verbose
595 if isdefined(integrator, :EEst)
596 EEst = ", and step error estimate = $(integrator.EEst)"
597 else
598 EEst = ""
599 end
600 @warn("At t=$(integrator.t), dt was forced below floating point epsilon $(integrator.dt)$EEst. Aborting. There is either an error in your model specification or the true solution is unstable (or the true solution can not be represented in the precision of $(eltype(integrator.u))).")
601 end
602 return ReturnCode.Unstable
603 end
604 end
605 if step_accepted &&
606 opts.unstable_check(integrator.dt, integrator.u, integrator.p, integrator.t)
607 if verbose
608 @warn("Instability detected. Aborting")
609 end
610 return ReturnCode.Unstable
611 end
612 if last_step_failed(integrator)
613 if verbose
614 @warn("Newton steps could not converge and algorithm is not adaptive. Use a lower dt.")
615 end
616 return ReturnCode.ConvergenceFailure
617 end
618 return ReturnCode.Success
619 end

718 struct TimeChoiceIterator{T, T2}
719 integrator::T
720 ts::T2
721 end
722
723 function Base.iterate(iter::TimeChoiceIterator, state = 1)
724 state > length(iter.ts) && return nothing
725 t = iter.ts[state]
726 integrator = iter.integrator
727 if isinplace(integrator.sol.prob)
728 tmp = first(get_tmp_cache(integrator))
729 if t == integrator.t
730 tmp .= integrator.u
731 elseif t < integrator.t
732 integrator(tmp, t)
733 else
734 step!(integrator, t - integrator.t)
735 integrator(tmp, t)
736 end
737 return (tmp, t), state + 1
738 else
739 if t == integrator.t
740 tmp = integrator.u
741 elseif t < integrator.t
742 tmp = integrator(t)
743 else
744 step!(integrator, t - integrator.t)
745 tmp = integrator(t)
746 end
747 return (tmp, t), state + 1
748 end
749 end

755 integrator isa AbstractSDEIntegrator) &&
756 integrator.iter > 0,

768 int_vars = interpret_vars(idxs, integrator.sol)

781 end

783 dims = length(int_vars[1])
784 for var in int_vars
785 @assert length(var) == dims

796 varsyms = variable_symbols(integrator)
797 @show plott

834 u_n(plot_timeseries, x[j], sol, plott, plot_timeseries))

848 end

859 end

883 function step!(integ::DEIntegrator, dt, stop_at_tdt = false)
884 (dt * integ.tdir) < 0 * oneunit(dt) && error("Cannot step backward.")
885 t = integ.t
886 next_t = t + dt
887 stop_at_tdt && add_tstop!(integ, next_t)
888 while integ.t * integ.tdir < next_t * integ.tdir
889 step!(integ)
890 integ.sol.retcode in (ReturnCode.Default, ReturnCode.Success) || break
891 end
892 end
