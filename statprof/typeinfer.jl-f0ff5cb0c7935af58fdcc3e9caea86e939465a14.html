51 Core.Compiler.Timing(mi_info, start_time, ...)
52
53 Internal type containing the timing result for running type inference on a single
54 MethodInstance.
55 """
56 struct Timing
57 mi_info::InferenceFrameInfo
58 start_time::UInt64
59 cur_start_time::UInt64
60 time::UInt64
61 children::Core.Array{Timing,1}
62 bt # backtrace collected upon initial entry to typeinf
63 end
64 Timing(mi_info, start_time, cur_start_time, time, children) = Timing(mi_info, start_time, cur_start_time, time, children, nothing)
65 Timing(mi_info, start_time) = Timing(mi_info, start_time, start_time, UInt64(0), Timing[])
66
67 _time_ns() = ccall(:jl_hrtime, UInt64, ()) # Re-implemented here because Base not yet available.
68
69 # We keep a stack of the Timings for each of the MethodInstances currently being timed.
70 # Since type inference currently operates via a depth-first search (during abstract
71 # evaluation), this vector operates like a call stack. The last node in _timings is the
72 # node currently being inferred, and its parent is directly before it, etc.
73 # Each Timing also contains its own vector for all of its children, so that the tree
74 # call structure through type inference is recorded. (It's recorded as a tree, not a graph,
75 # because we create a new node for duplicates.)
76 const _timings = Timing[]
77 # ROOT() is an empty function used as the top-level Timing node to measure all time spent
78 # *not* in type inference during a given recording trace. It is used as a "dummy" node.
79 function ROOT() end
80 const ROOTmi = Core.Compiler.specialize_method(
81 first(Core.Compiler.methods(ROOT)), Tuple{typeof(ROOT)}, Core.svec())
82 """
83 Core.Compiler.reset_timings()
84
85 Empty out the previously recorded type inference timings (`Core.Compiler._timings`), and
86 start the ROOT() timer again. `ROOT()` measures all time spent _outside_ inference.
87 """
88 function reset_timings()
89 empty!(_timings)
90 push!(_timings, Timing(
91 # The MethodInstance for ROOT(), and default empty values for other fields.
92 InferenceFrameInfo(ROOTmi, 0x0, Core.Compiler.VarState[], Any[Core.Const(ROOT)], 1),
93 _time_ns()))
94 return nothing
95 end
96 reset_timings()

98 # (This is split into a function so that it can be called both in this module, at the top
99 # of `enter_new_timer()`, and once at the Very End of the operation, by whoever started
100 # the operation and called `reset_timings()`.)
101 # NOTE: the @inline annotations here are not to make it faster, but to reduce the gap between
102 # timer manipulations and the tasks we're timing.
103 @inline function close_current_timer()
104 stop_time = _time_ns()
105 parent_timer = _timings[end]
106 accum_time = stop_time - parent_timer.cur_start_time
107
108 # Add in accum_time ("modify" the immutable struct)
109 @inbounds begin
110 _timings[end] = Timing(
111 parent_timer.mi_info,
112 parent_timer.start_time,
113 parent_timer.cur_start_time,
114 parent_timer.time + accum_time,
115 parent_timer.children,
116 parent_timer.bt,
117 )
118 end
119 return nothing
120 end

122 @inline function enter_new_timer(frame)
123 # Very first thing, stop the active timer: get the current time and add in the
124 # time since it was last started to its aggregate exclusive time.
125 close_current_timer()
126
127 mi_info = _typeinf_identifier(frame)
128
129 # Start the new timer right before returning
130 push!(_timings, Timing(mi_info, UInt64(0)))
131 len = length(_timings)
132 new_timer = @inbounds _timings[len]
133 # Set the current time _after_ appending the node, to try to exclude the
134 # overhead from measurement.
135 start = _time_ns()
136
137 @inbounds begin
138 _timings[len] = Timing(
139 new_timer.mi_info,
140 start,
141 start,
142 new_timer.time,
143 new_timer.children,
144 )
145 end
146
147 return nothing
148 end
149
150 # _expected_frame_ is not needed within this function; it is used in the `@assert`, to
151 # assert that indeed we are always returning to a parent after finishing all of its
152 # children (that is, asserting that inference proceeds via depth-first-search).
153 @inline function exit_current_timer(_expected_frame_)
154 # Finish the new timer
155 stop_time = _time_ns()
156
157 expected_mi_info = _typeinf_identifier(_expected_frame_)
158
159 # Grab the new timer again because it might have been modified in _timings
160 # (since it's an immutable struct)
161 # And remove it from the current timings stack
162 new_timer = pop!(_timings)
163 Core.Compiler.@assert new_timer.mi_info.mi === expected_mi_info.mi
164
165 # Prepare to unwind one level of the stack and record in the parent
166 parent_timer = _timings[end]
167
168 accum_time = stop_time - new_timer.cur_start_time
169 # Add in accum_time ("modify" the immutable struct)
170 new_timer = Timing(
171 new_timer.mi_info,
172 new_timer.start_time,
173 new_timer.cur_start_time,
174 new_timer.time + accum_time,
175 new_timer.children,
176 parent_timer.mi_info.mi === ROOTmi ? backtrace() : nothing,
177 )
178 # Record the final timing with the original parent timer
179 push!(parent_timer.children, new_timer)
180
181 # And finally restart the parent timer:
182 len = length(_timings)
183 @inbounds begin
184 _timings[len] = Timing(
185 parent_timer.mi_info,
186 parent_timer.start_time,
187 _time_ns(),
188 parent_timer.time,
189 parent_timer.children,
190 parent_timer.bt,
191 )
192 end
193
194 return nothing
195 end

245 function _typeinf(interp::AbstractInterpreter, frame::InferenceState)
246 interp = switch_from_irinterp(interp)
247 typeinf_nocycle(interp, frame) || return false # frame is now part of a higher cycle
248 # with no active ip's, frame is done
249 frames = frame.callers_in_cycle
250 isempty(frames) && push!(frames, frame)
251 valid_worlds = WorldRange()
252 for caller in frames
253 @assert !(caller.dont_work_on_me)
254 caller.dont_work_on_me = true
255 # might might not fully intersect these earlier, so do that now
256 valid_worlds = intersect(caller.valid_worlds, valid_worlds)
257 end
258 for caller in frames
259 caller.valid_worlds = valid_worlds
260 finish(caller, interp)
261 end
262 # collect results for the new expanded frame
263 results = Tuple{InferenceResult, Vector{Any}, Bool}[
264 ( frames[i].result,
265 frames[i].stmt_edges[1]::Vector{Any},
266 frames[i].cached )
267 for i in 1:length(frames) ]
268 empty!(frames)
269 for (caller, _, _) in results
270 opt = caller.src
271 if opt isa OptimizationState{typeof(interp)} # implies `may_optimize(interp) === true`
272 optimize(interp, opt, caller)
273 end
274 end
275 for (caller, edges, cached) in results
276 valid_worlds = caller.valid_worlds
277 if last(valid_worlds) >= get_world_counter()
278 # if we aren't cached, we don't need this edge
279 # but our caller might, so let's just make it anyways
280 store_backedges(caller, edges)
281 end
282 if cached
283 cache_result!(interp, caller)
284 end
285 finish!(interp, caller)
286 end
287 return true
288 end

440 function adjust_effects(sv::InferenceState)
441 ipo_effects = sv.ipo_effects
442
443 # refine :consistent-cy effect using the return type information
444 # TODO this adjustment tries to compromise imprecise :consistent-cy information,
445 # that is currently modeled in a flow-insensitive way: ideally we want to model it
446 # with a proper dataflow analysis instead
447 rt = sv.bestguess
448 if ipo_effects.noinbounds && rt === Bottom
449 # always throwing an error counts or never returning both count as consistent
450 ipo_effects = Effects(ipo_effects; consistent=ALWAYS_TRUE)
451 end
452 if is_inaccessiblemem_or_argmemonly(ipo_effects) && all(1:narguments(sv, #=include_va=#true)) do i::Int
453 return is_mutation_free_argtype(sv.slottypes[i])
454 end
455 ipo_effects = Effects(ipo_effects; inaccessiblememonly=ALWAYS_TRUE)
456 end
457 if is_consistent_if_notreturned(ipo_effects) && is_identity_free_argtype(rt)
458 # in a case when the :consistent-cy here is only tainted by mutable allocations
459 # (indicated by `CONSISTENT_IF_NOTRETURNED`), we may be able to refine it if the return
460 # type guarantees that the allocations are never returned
461 consistent = ipo_effects.consistent & ~CONSISTENT_IF_NOTRETURNED
462 ipo_effects = Effects(ipo_effects; consistent)
463 end
464 if is_consistent_if_inaccessiblememonly(ipo_effects)
465 if is_inaccessiblememonly(ipo_effects)
466 consistent = ipo_effects.consistent & ~CONSISTENT_IF_INACCESSIBLEMEMONLY
467 ipo_effects = Effects(ipo_effects; consistent)
468 elseif is_inaccessiblemem_or_argmemonly(ipo_effects)
469 else # `:inaccessiblememonly` is already tainted, there will be no chance to refine this
470 ipo_effects = Effects(ipo_effects; consistent=ALWAYS_FALSE)
471 end
472 end
473 if is_effect_free_if_inaccessiblememonly(ipo_effects)
474 if is_inaccessiblememonly(ipo_effects)
475 effect_free = ipo_effects.effect_free & ~EFFECT_FREE_IF_INACCESSIBLEMEMONLY
476 ipo_effects = Effects(ipo_effects; effect_free)
477 elseif is_inaccessiblemem_or_argmemonly(ipo_effects)
478 else # `:inaccessiblememonly` is already tainted, there will be no chance to refine this
479 ipo_effects = Effects(ipo_effects; effect_free=ALWAYS_FALSE)
480 end
481 end
482
483 # override the analyzed effects using manually annotated effect settings
484 def = sv.linfo.def
485 if isa(def, Method)
486 override = decode_effects_override(def.purity)
487 if is_effect_overridden(override, :consistent)
488 ipo_effects = Effects(ipo_effects; consistent=ALWAYS_TRUE)
489 end
490 if is_effect_overridden(override, :effect_free)
491 ipo_effects = Effects(ipo_effects; effect_free=ALWAYS_TRUE)
492 end
493 if is_effect_overridden(override, :nothrow)
494 ipo_effects = Effects(ipo_effects; nothrow=true)
495 end
496 if is_effect_overridden(override, :terminates_globally)
497 ipo_effects = Effects(ipo_effects; terminates=true)
498 end
499 if is_effect_overridden(override, :notaskstate)
500 ipo_effects = Effects(ipo_effects; notaskstate=true)
501 end
502 if is_effect_overridden(override, :inaccessiblememonly)
503 ipo_effects = Effects(ipo_effects; inaccessiblememonly=ALWAYS_TRUE)
504 end
505 end
506
507 return ipo_effects
508 end

510 # inference completed on `me`
511 # update the MethodInstance
512 function finish(me::InferenceState, interp::AbstractInterpreter)
513 # prepare to run optimization passes on fulltree
514 s_edges = me.stmt_edges[1]
515 if s_edges === nothing
516 s_edges = me.stmt_edges[1] = []
517 end
518 for edges in me.stmt_edges
519 edges === nothing && continue
520 edges === s_edges && continue
521 append!(s_edges, edges)
522 empty!(edges)
523 end
524 if me.src.edges !== nothing
525 append!(s_edges, me.src.edges::Vector)
526 me.src.edges = nothing
527 end
528 # inspect whether our inference had a limited result accuracy,
529 # else it may be suitable to cache
530 bestguess = me.bestguess = cycle_fix_limited(me.bestguess, me)
531 limited_ret = bestguess isa LimitedAccuracy
532 limited_src = false
533 if !limited_ret
534 gt = me.ssavaluetypes
535 for j = 1:length(gt)
536 gt[j] = gtj = cycle_fix_limited(gt[j], me)
537 if gtj isa LimitedAccuracy && me.parent !== nothing
538 limited_src = true
539 break
540 end
541 end
542 end
543 if limited_ret
544 # a parent may be cached still, but not this intermediate work:
545 # we can throw everything else away now
546 me.result.src = nothing
547 me.cached = false
548 set_inlineable!(me.src, false)
549 unlock_mi_inference(interp, me.linfo)
550 elseif limited_src
551 # a type result will be cached still, but not this intermediate work:
552 # we can throw everything else away now
553 me.result.src = nothing
554 set_inlineable!(me.src, false)
555 else
556 # annotate fulltree with type information,
557 # either because we are the outermost code, or we might use this later
558 doopt = (me.cached || me.parent !== nothing)
559 recompute_cfg = type_annotate!(interp, me, doopt)
560 if doopt && may_optimize(interp)
561 me.result.src = OptimizationState(me, interp, recompute_cfg)
562 else
563 me.result.src = me.src::CodeInfo # stash a convenience copy of the code (e.g. for reflection)
564 end
565 end
566 me.result.valid_worlds = me.valid_worlds
567 me.result.result = bestguess
568 me.ipo_effects = me.result.ipo_effects = adjust_effects(me)
569 validate_code_in_debug_mode(me.linfo, me.src, "inferred")
570 nothing
571 end

592 function record_slot_assign!(sv::InferenceState)
593 # look at all assignments to slots
594 # and union the set of types stored there
595 # to compute a lower bound on the storage required
596 body = sv.src.code::Vector{Any}
597 slottypes = sv.slottypes::Vector{Any}
598 ssavaluetypes = sv.ssavaluetypes
599 for i = 1:length(body)
600 expr = body[i]
601 # find all reachable assignments to locals
602 if was_reached(sv, i) && isexpr(expr, :(=))
603 lhs = expr.args[1]
604 if isa(lhs, SlotNumber)
605 typ = ssavaluetypes[i]
606 @assert typ !== NOT_FOUND "active slot in unreached region"
607 vt = widenconst(typ)
608 if vt !== Bottom
609 id = slot_id(lhs)
610 otherTy = slottypes[id]
611 if otherTy === Bottom
612 slottypes[id] = vt
613 elseif otherTy === Any
614 slottypes[id] = Any
615 else
616 slottypes[id] = tmerge(otherTy, vt)
617 end
618 end
619 end
620 end
621 end
622 sv.src.slottypes = slottypes
623 return nothing
624 end

633 function annotate_slot_load!(interp::AbstractInterpreter, undefs::Vector{Bool}, idx::Int, sv::InferenceState, @nospecialize x)
634 if isa(x, SlotNumber)
635 id = slot_id(x)
636 pc = find_dominating_assignment(id, idx, sv)
637 if pc === nothing
638 block = block_for_inst(sv.cfg, idx)
639 state = sv.bb_vartables[block]::VarTable
640 vt = state[id]
641 undefs[id] |= vt.undef
642 typ = widenslotwrapper(ignorelimited(vt.typ))
643 else
644 typ = sv.ssavaluetypes[pc]
645 @assert typ !== NOT_FOUND "active slot in unreached region"
646 end
647 # add type annotations where needed
648 if !(typeinf_lattice(interp), sv.slottypes[id], typ)
649 return TypedSlot(id, typ)
650 end
651 return x
652 elseif isa(x, Expr)
653 head = x.head
654 i0 = 1
655 if is_meta_expr_head(head) || head === :const
656 return x
657 end
658 if head === :(=) || head === :method
659 i0 = 2
660 end
661 for i = i0:length(x.args)
662 x.args[i] = annotate_slot_load!(interp, undefs, idx, sv, x.args[i])
663 end
664 return x
665 elseif isa(x, ReturnNode) && isdefined(x, :val)
666 return ReturnNode(annotate_slot_load!(interp, undefs, idx, sv, x.val))
667 elseif isa(x, GotoIfNot)
668 return GotoIfNot(annotate_slot_load!(interp, undefs, idx, sv, x.cond), x.dest)

673 # find the dominating assignment to the slot `id` in the block containing statement `idx`,
674 # returns `nothing` otherwise
675 function find_dominating_assignment(id::Int, idx::Int, sv::InferenceState)
676 block = block_for_inst(sv.cfg, idx)
677 for pc in reverse(sv.cfg.blocks[block].stmts) # N.B. reverse since the last assignment is dominating this block
678 pc < idx || continue # N.B. needs pc idx as `id` can be assigned at `idx`
679 stmt = sv.src.code[pc]
680 isexpr(stmt, :(=)) || continue
681 lhs = stmt.args[1]
682 isa(lhs, SlotNumber) || continue
683 slot_id(lhs) == id || continue
684 return pc
685 end
686 return nothing

689 # annotate types of all symbols in AST, preparing for optimization
690 function type_annotate!(interp::AbstractInterpreter, sv::InferenceState, run_optimizer::Bool)
691 # widen `Conditional`s from `slottypes`
692 slottypes = sv.slottypes
693 for i = 1:length(slottypes)
694 slottypes[i] = widenconditional(slottypes[i])
695 end
696
697 # compute the required type for each slot
698 # to hold all of the items assigned into it
699 record_slot_assign!(sv)
700
701 record_bestguess!(sv)
702
703 # annotate variables load types
704 # remove dead code optimization
705 # and compute which variables may be used undef
706 stmt_info = sv.stmt_info
707 src = sv.src
708 body = src.code
709 nexpr = length(body)
710 codelocs = src.codelocs
711 ssavaluetypes = sv.ssavaluetypes
712 ssaflags = src.ssaflags
713 slotflags = src.slotflags
714 nslots = length(slotflags)
715 undefs = fill(false, nslots)
716 any_unreachable = false
717
718 # this statement traversal does five things:
719 # 1. introduce temporary `TypedSlot`s that are supposed to be replaced with -nodes later
720 # 2. mark used-undef slots (required by the `slot2reg` conversion)
721 # 3. mark unreached statements for a bulk code deletion (see issue #7836)
722 # 4. widen slot wrappers (`Conditional` and `MustAlias`) and remove `NOT_FOUND` from `ssavaluetypes`
723 # NOTE because of this, `was_reached` will no longer be available after this point
724 # 5. eliminate GotoIfNot if either branch target is unreachable
725 changemap = nothing # initialized if there is any dead region
726 for i = 1:nexpr
727 expr = body[i]
728 if was_reached(sv, i)
729 if run_optimizer
730 if isa(expr, GotoIfNot) && widenconst(argextype(expr.cond, src, sv.sptypes)) === Bool
731 # 5: replace this live GotoIfNot with:
732 # - GotoNode if the fallthrough target is unreachable
733 # - no-op if the branch target is unreachable
734 if !was_reached(sv, i+1)
735 expr = GotoNode(expr.dest)
736 elseif !was_reached(sv, expr.dest)
737 expr = nothing
738 end
739 end
740 end
741 body[i] = annotate_slot_load!(interp, undefs, i, sv, expr) # 1&2
742 ssavaluetypes[i] = widenslotwrapper(ssavaluetypes[i]) # 4
743 else # i.e. any runtime execution will never reach this statement
744 any_unreachable = true
745 if is_meta_expr(expr) # keep any lexically scoped expressions
746 ssavaluetypes[i] = Any # 4
747 else
748 ssavaluetypes[i] = Bottom # 4
749 body[i] = Const(expr) # annotate that this statement actually is dead
750 end
751 end
752 end
753
754 # finish marking used-undef variables
755 for j = 1:nslots
756 if undefs[j]
757 slotflags[j] |= SLOT_USEDUNDEF | SLOT_STATICUNDEF
758 end
759 end
760
761 return any_unreachable
762 end

765 # will now be added to a's cycle
766 function union_caller_cycle!(a::InferenceState, b::InferenceState)
767 callers_in_cycle = b.callers_in_cycle
768 b.parent = a.parent
769 b.callers_in_cycle = a.callers_in_cycle
770 contains_is(a.callers_in_cycle, b) || push!(a.callers_in_cycle, b)
771 if callers_in_cycle !== a.callers_in_cycle
772 for caller in callers_in_cycle
773 if caller !== b
774 caller.parent = a.parent
775 caller.callers_in_cycle = a.callers_in_cycle
776 push!(a.callers_in_cycle, caller)
777 end
778 end
779 end
780 return

796 parent = frame_parent(parent::IRInterpretationState)
797 end

811 # Walk through `mi`'s upstream call chain, starting at `parent`. If a parent
812 # frame matching `mi` is encountered, then there is a cycle in the call graph
813 # (i.e. `mi` is a descendant callee of itself). Upon encountering this cycle,
814 # we "resolve" it by merging the call chain, which entails unioning each intermediary
815 # frame's `callers_in_cycle` field and adding the appropriate backedges. Finally,
816 # we return `mi`'s pre-existing frame. If no cycles are found, `nothing` is
817 # returned instead.
818 function resolve_call_cycle!(interp::AbstractInterpreter, mi::MethodInstance, parent::AbsIntState)
819 # TODO (#48913) implement a proper recursion handling for irinterp:
820 # This works just because currently the `:terminate` condition guarantees that
821 # irinterp doesn't fail into unresolved cycles, but it's not a good solution.
822 # We should revisit this once we have a better story for handling cycles in irinterp.
823 isa(parent, InferenceState) || return false
824 frame = parent
825 uncached = false
826 while isa(frame, InferenceState)
827 uncached |= !is_cached(frame) # ensure we never add an uncached frame to a cycle
828 if is_same_frame(interp, mi, frame)
829 if uncached
830 # our attempt to speculate into a constant call lead to an undesired self-cycle
831 # that cannot be converged: poison our call-stack (up to the discovered duplicate frame)
832 # with the limited flag and abort (set return type to Any) now
833 poison_callstack!(parent, frame)
834 return true
835 end
836 merge_call_chain!(interp, parent, frame, frame)
837 return frame
838 end
839 for caller in callers_in_cycle(frame)
840 if is_same_frame(interp, mi, caller)
841 if uncached
842 poison_callstack!(parent, frame)
843 return true
844 end
845 merge_call_chain!(interp, parent, frame, caller)
846 return caller
847 end
848 end
849 frame = frame_parent(frame)
850 end
851 return false
852 end

869 # compute (and cache) an inferred AST and return the current best estimate of the result type
870 function typeinf_edge(interp::AbstractInterpreter, method::Method, @nospecialize(atype), sparams::SimpleVector, caller::AbsIntState)
871 mi = specialize_method(method, atype, sparams)::MethodInstance
872 code = get(code_cache(interp), mi, nothing)
873 if code isa CodeInstance # return existing rettype if the code is already inferred
874 inferred = @atomic :monotonic code.inferred
875 if inferred === nothing && is_stmt_inline(get_curr_ssaflag(caller))
876 # we already inferred this edge before and decided to discard the inferred code,
877 # nevertheless we re-infer it here again and keep it around in the local cache
878 # since the inliner will request to use it later
879 cache = :local
880 else
881 effects = ipo_effects(code)
882 update_valid_age!(caller, WorldRange(min_world(code), max_world(code)))
883 rettype = code.rettype
884 if isdefined(code, :rettype_const)
885 rettype_const = code.rettype_const
886 # the second subtyping/egal conditions are necessary to distinguish usual cases
887 # from rare cases when `Const` wrapped those extended lattice type objects
888 if isa(rettype_const, Vector{Any}) && !(Vector{Any} <: rettype)
889 rettype = PartialStruct(rettype, rettype_const)
890 elseif isa(rettype_const, PartialOpaque) && rettype <: Core.OpaqueClosure
891 rettype = rettype_const
892 elseif isa(rettype_const, InterConditional) && rettype !== InterConditional
893 rettype = rettype_const
894 elseif isa(rettype_const, InterMustAlias) && rettype !== InterMustAlias
895 rettype = rettype_const
896 else
897 rettype = Const(rettype_const)
898 end
899 end
900 return EdgeCallResult(rettype, mi, effects)
901 end
902 else
903 cache = :global # cache edge targets by default
904 end
905 if ccall(:jl_get_module_infer, Cint, (Any,), method.module) == 0 && !generating_sysimg()
906 add_remark!(interp, caller, "Inference is disabled for the target module")
907 return EdgeCallResult(Any, nothing, Effects())
908 end
909 if !is_cached(caller) && frame_parent(caller) === nothing
910 # this caller exists to return to the user
911 # (if we asked resolve_call_cycle!, it might instead detect that there is a cycle that it can't merge)
912 frame = false
913 else
914 frame = resolve_call_cycle!(interp, mi, caller)
915 end
916 if frame === false
917 # completely new
918 lock_mi_inference(interp, mi)
919 result = InferenceResult(mi, typeinf_lattice(interp))
920 frame = InferenceState(result, cache, interp) # always use the cache for edge targets
921 if frame === nothing
922 add_remark!(interp, caller, "Failed to retrieve source")
923 # can't get the source for this, so we know nothing
924 unlock_mi_inference(interp, mi)
925 return EdgeCallResult(Any, nothing, Effects())
926 end
927 if is_cached(caller) || frame_parent(caller) !== nothing # don't involve uncached functions in cycle resolution
928 frame.parent = caller
929 end
930 typeinf(interp, frame)
931 update_valid_age!(caller, frame.valid_worlds)
932 edge = is_inferred(frame) ? mi : nothing
933 return EdgeCallResult(frame.bestguess, edge, frame.ipo_effects) # effects are adjusted already within `finish`
934 elseif frame === true
935 # unresolvable cycle
936 return EdgeCallResult(Any, nothing, Effects())
937 end
938 # return the current knowledge about this cycle
939 frame = frame::InferenceState
940 update_valid_age!(caller, frame.valid_worlds)
941 return EdgeCallResult(frame.bestguess, nothing, adjust_effects(frame))
942 end
