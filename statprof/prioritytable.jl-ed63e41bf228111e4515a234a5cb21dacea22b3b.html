1 """
2 Dynamic table data structure to store and update priorities
3
4 Implementation
5 Stores ids that identify the priorities. Each group stores a range of priority ids.
6 The basic design assumes a lower-most group of "zero" priorities, and a second
7 group storing all non-zero priorities that are < a `minpriority`. All other groups
8 store priorities within consecutive ranges. The `minpriority` is taken fixed at creation,
9 but the `maxpriority` will increase dynamically based on inserted / updated priorities.
10 The ranges are assumed to be powers of two:
11 bin 1 = {0},
12 bin 2 = (0,`minpriority`),
13 bin 3 = [`minpriority`,`2*minpriority`)...
14 bin N = [`.5*maxpriority`,`maxpriority`)
15 *Assumes* the `priortogid` function that maps priorities to groups maps the upper end of
16 the interval to the next group. i.e. maxpriority -> N+1
17 """

97 """
98 Setup table from a vector of priorities. The id
99 of a priority is its position within this vector.
100 """
101 function PriorityTable(priortogid::Function, priorities::AbstractVector, minpriority,
102 maxpriority)
103 numgroups = priortogid(maxpriority)
104 numgroups -= one(typeof(numgroups))
105 pidtype = typeof(numgroups)
106 ptype = eltype(priorities)
107 groups = Vector{PriorityGroup{ptype, Vector{pidtype}}}()
108 pidtogroup = Vector{Tuple{Int, Int}}(undef, length(priorities))
109 gsum = zero(ptype)
110 gsums = zeros(ptype, numgroups)
111
112 # create the groups, {0}, (0,minpriority), [minpriority,2*minpriority)...
113 push!(groups, PriorityGroup{pidtype}(zero(ptype)))
114 gmaxprior = minpriority
115 for i in 2:numgroups
116 push!(groups, PriorityGroup{pidtype}(gmaxprior))
117 gmaxprior *= 2
118 end
119 #@assert 2*maxpriority <= gmaxprior "gmaxprior = $gmaxprior, maxpriority=$maxpriority"
120 gmaxprior = groups[end].maxpriority
121
122 pt = PriorityTable(minpriority, gmaxprior, groups, gsums, gsum, pidtogroup, priortogid)
123
124 # insert priority ids into the groups
125 for (pid, priority) in enumerate(priorities)
126 insert!(pt, pid, priority)
127 end

190 function update!(pt::PriorityTable, pid, oldpriority, newpriority)
191 @unpack maxpriority, groups, gsums, pidtogroup, priortogid = pt
192
193 oldgid = priortogid(oldpriority)
194 newgid = priortogid(newpriority)
195
196 # expand the table if necessary
197 if newpriority >= maxpriority
198 padtable!(pt, pid, newpriority)
199 end
200
201 # update the global priority sum
202 pdiff = newpriority - oldpriority
203 pt.gsum += pdiff
204
205 if oldgid == newgid
206 # update the group priority too
207 @inbounds gsums[newgid] += pdiff
208 else
209 @inbounds begin
210 # location in oldgid of pid to move
211 pididx = pidtogroup[pid][2]
212
213 # remove pid from old group
214 movedpid = remove!(groups[oldgid], pididx)
215
216 # update position in group of the pid that swapped places with the removed one
217 pidtogroup[movedpid] = (oldgid, pididx)
218
219 # insert the updated pid back and store it's new position in the group
220 pididx = insert!(groups[newgid], pid)
221 pidtogroup[pid] = (newgid, pididx)
222
223 # update sums, special case if group empty to avoid FP error in running sums
224 grpsz = groups[oldgid].numpids
225 gsums[oldgid] = (grpsz == zero(grpsz)) ? zero(oldpriority) :
226 gsums[oldgid] - oldpriority
227 gsums[newgid] += newpriority
228 end
229 end
